Fixing the 'each line' problem

Shell script


IFS=$'\n'
PROCESSES=`ps aux`
for process in $PROCESSES
do
  if [[ $process =~ ^$USER.*$ ]]
  then
    echo $process
  fi
done

xonsh


def ls_files_home():
    return $(ls).split('\n')[:-1]

ls_files_home()

def ls_files(dir):
    return $(ls @(dir)).split('\n')[:-1]

ls_files('/')
ls_files($HOME)
ls_files(${'HOME'})

Listing processes

xonsh


procs_info = [proc.split() for proc in $(ps aux).split('\n')[:-1]]
headers = procs_info[0]
procs = []
for proc_info in procs_info[1:]:
    proc_dict = {}
    for (index, header) in enumerate(headers):
        proc_dict[header] = proc_info[index]
    procs.append(proc_dict)

procs
procs[0]
echo @(procs)

Building a dictionary from the command's output
